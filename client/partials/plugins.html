Add jsTree plugins

Include jsTree.directive.jsin your index.htmland add any of the below configurations to load a Static Tree

Add Checkbox plugin

Set tree-plugins="checkbox" 

<js-tree tree-plugins="checkbox" tree-data="json" tree-src="../jstree/data/tree-json.json"></js-tree>

Angular Directive Integration 
var app = angular.module('treeDemo', ['jsTree.directive'])

Add Custom Contextmenu plugin

Set tree-plugins="contextmenu"and pass the context menu object via tree-contextmenu="scopeObjName" 

<js-tree tree-plugins="contextmenu" tree-contextmenu="contextMenu" tree-data="json" tree-src="../jstree/data/tree-json.json"></js-tree>

Angular Directive Integration 
var app = angular.module('treeDemo', ['jsTree.directive'])


Scope object 
$scope.contextMenu = {
      "Menu 1": {
        "label": "Menu 1",
        "action": function(obj) {
          console.log(obj);
          alert("You clicked " + obj.item.label);
        }
      },
      "Menu 2": {
        "label": "Menu 2",
        "action": function(obj) {
          console.log(obj);
          alert("You clicked " + obj.item.label);
        }
      }
    };

Add Checkbox & Context menu plugins

Set tree-plugins="checkbox,contextmenu" 

<js-tree tree-plugins="checkbox,contextmenu" tree-contextmenu="/config/contextmenu.json" tree-data="json" tree-src="../jstree/data/tree-json.json"></js-tree>

Angular Directive Integration 
var app = angular.module('treeDemo', ['jsTree.directive'])

Demo

Add Types plugin

Set tree-plugins="types"and pass the config object via tree-types="typesConfig" 

<js-tree tree-plugins="types" tree-types="typesConfig" tree-data="json" tree-src="../jstree/data/tree-json.json"></js-tree>

Angular Directive Integration 
var app = angular.module('treeDemo', ['jsTree.directive'])


Scope object 
$scope.typesConfig = {
      "default": {
        "icon": "glyphicon glyphicon-flash"
      },
      "demo": {
        "icon": "glyphicon glyphicon-ok"
      }
    }
